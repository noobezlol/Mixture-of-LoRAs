    new_string = ''

    return new_string[::-1] + string

def binary_search(list, item):
    left = 0
    right = len(list) - 1

    while left <= right:
        mid = (left + right) // 2
        if list[mid] == item:
            return mid
        elif list[mid] < item:
            left = mid + 1
        else:
            right = mid - 1

    return None

    card_number = [int(x) for x in str(card_number)]
    # Reverse the list
    card_number.reverse()
    # Multiply every second digit by 2
    for i in range(1, len(card_number), 2):
        card_number[i] *= 2
    # Sum all digits
    card_number = sum(card_number)
    # Check if the sum is divisible by 10
    if card_number % 10 == 0:
        return True
    else:
        return False

def flatten_dict(d, parent_key='', sep='.'):
    items = []
    for k, v in d.items():
        new_key = parent_key + sep + k if parent_key else k
        if isinstance(v, dict):
            items.extend(flatten_dict(v, new_key, sep=sep).items())
        else:
            items.append((new_key, v))
    return dict(items)

Given nums = [2, 7, 11, 15], target = 9,

Because nums[0] + nums[1] = 2 + 7 = 9,
return [0, 1].
"""

class Solution:
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                if nums[i] + nums[j] == target:
                    return [i, j]

# Definition for singly-linked list.
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None

class Solution:
    def addTwoNumbers(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode

                carry += l2.val
                l2 = l2.next
            curr.next = ListNode(carry % 10)
            curr = curr.next
            carry //= 10
        return head.next

import math

def sieve_of_eratosthenes(n):
    """
    Returns a list of all primes up to n.
    """
    # Create a list of booleans to represent the numbers from 2 to n
    # True means the number is prime
    # False means the number is not prime
    sieve = [True] * (n + 1)

    # Set the first two numbers to False because they are not prime
    sieve[0] = False
    sieve[1] = False

    # Iterate through the numbers from 2 to n
    for i in range(2, n + 1):
        # If the number is prime, mark all multiples of that number as not prime
        if sieve[i]:
            for j in range(i * i, n + 1, i):
                sieve[j] = False

    # Return the list of numbers that are prime
    return [i for i in range(2, n + 1) if sieve[i]]

    # Test the function
    print(sieve_of_eratosthenes(100))

import os
import shutil
import errno

def delete_file(file_path):
    try:
        if os.path.exists(file_path):
            os.remove(file_path)
        else:
            print("The file does not exist")
    except OSError as e:
        if e.errno == errno.EACCES:
            print("Permission denied")
        elif e.errno == errno.EISDIR:
            print("Cannot delete a directory")
        else:
            print("Error: %s - %s." % (e.filename, e.strerror))

def delete_dir(dir_path):
    try:
        if os.path.exists(dir_path):
            shutil.rmtree(dir_path)
        else:
            print("The directory does not exist")
    except OSError as e:
        if e.errno == errno.EACCES:
            print("Permission denied")
        elif e.errno == errno.ENOTEMPTY:
            print("Directory is not empty")
        else:
            print("Error: %s - %s." % (e.filename, e.strerror))

delete_file("test.txt")
delete_dir("test")

def average(lst):
    """
    Calculate the average of a list.
    """
    if lst is None or len(lst) == 0:
        return None
    if any(isinstance(x, str) for x in lst):
        return None
    return sum(lst) / len(lst)

def merge_dicts(dict1, dict2):
    for key in dict2:
        if key in dict1:
            if type(dict1[key]) == list:
                dict1[key].append(dict2[key])
            else:
                dict1[key] = [dict1[key], dict2[key]]
        else:
            dict1[key] = dict2[key]
    return dict1
